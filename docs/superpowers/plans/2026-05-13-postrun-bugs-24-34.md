```python
# Gemini quota breaker — cycle-scoped (no persistence across process restarts).
# Once N consecutive RESOURCE_EXHAUSTED errors hit, short-circuit further grounded
# calls for the remainder of the cycle so we don't burn ~24 doomed calls per ticker
# during a quota outage. Reset to closed on the next process start or any success.
GEMINI_QUOTA_FAILURES_TO_TRIP = 4  # one full Phase-1 sweep is 4 sensor agents


class GeminiQuotaBreaker:
    """Process-lifetime circuit breaker for Gemini quota outages."""

    def __init__(self) -> None:
        self._consecutive_quota_failures = 0
        self._tripped = False

    def tripped(self) -> bool:
        return self._tripped

    def record_quota_failure(self) -> None:
        self._consecutive_quota_failures += 1
        if self._consecutive_quota_failures >= GEMINI_QUOTA_FAILURES_TO_TRIP:
            if not self._tripped:
                logger.warning(
                    "[Gemini breaker] tripped after %d consecutive RESOURCE_EXHAUSTED "
                    "errors; subsequent grounded calls will short-circuit this cycle.",
                    self._consecutive_quota_failures,
                )
            self._tripped = True

    def record_non_quota_retryable(self) -> None:
        # Other transient errors (503, timeouts) do not advance the quota counter.
        pass

    def record_success(self) -> None:
        if self._consecutive_quota_failures or self._tripped:
            logger.info("[Gemini breaker] resetting after successful call.")
        self._consecutive_quota_failures = 0
        self._tripped = False


# Shared module-level breaker, reset on process start.
_gemini_breaker = GeminiQuotaBreaker()


def is_quota_error(e: Exception) -> bool:
    msg = str(e)
    return "RESOURCE_EXHAUSTED" in msg or "429" in msg or "quota" in msg.lower()
```

```python
# Markers in agent error stubs that indicate retrying is pointless.
_NON_RECOVERABLE_ERROR_MARKERS = (
    "RESOURCE_EXHAUSTED",
    "Gemini quota breaker tripped",
    "[SHORT INPUT DETECTED:",  # input problem, not API problem
)


def _is_non_recoverable_failure(report: Optional[str]) -> bool:
    if not isinstance(report, str):
        return False
    return any(marker in report for marker in _NON_RECOVERABLE_ERROR_MARKERS)


def _retry_failed_phase1_agents(reports, retry_prompt_map, call_fn, state):
    """Retry Phase 1 agents that produced error stubs, with two guards:

    1. Skip the entire retry pass if the Gemini quota breaker is tripped.
    2. Skip any agent whose prior failure is non-recoverable (RESOURCE_EXHAUSTED,
       input too short, etc.) — retrying won't change the outcome.
    """
    if _gemini_breaker.tripped():
        logger.info("[Phase 1 Retry] Skipped (Gemini breaker tripped).")
        return
    for key, (prompt, agent_label) in retry_prompt_map.items():
        current = reports.get(key)
        if _is_real_report(current):
            continue
        if _is_non_recoverable_failure(current):
            logger.info(
                "[Phase 1 Retry] %s skipped — prior failure non-recoverable.", agent_label
            )
            continue
        print(f"  > [Phase 1 Retry] {agent_label} failed first pass; retrying once...")
        try:
            retry_result = call_fn(prompt, agent_label, state)
            if _is_real_report(retry_result):
                reports[key] = retry_result
                print(f"  > [Phase 1 Retry] {agent_label} succeeded on retry.")
            else:
                print(f"  > [Phase 1 Retry] {agent_label} still failing after retry.")
        except Exception as e:
            logger.warning(f"[Phase 1 Retry] {agent_label} retry raised: {e}")
```

```python
"""One-shot cleanup of stray TEST_ rows from the production DB.

Run manually after the dedupe migration. Safe to re-run (idempotent)."""

import os
import sqlite3
import sys

DB_PATH = os.environ.get("DB_PATH", "subscribers.db")


def main() -> int:
    if not os.path.exists(DB_PATH):
        print(f"DB not found at {DB_PATH}; nothing to clean.")
        return 0
    conn = sqlite3.connect(DB_PATH)
    try:
        cur = conn.execute(
            r"SELECT id, symbol, timestamp FROM decision_points "
            r"WHERE symbol LIKE 'TEST\_%' ESCAPE '\'"
        )
        rows = cur.fetchall()
        if not rows:
            print("No TEST_ rows to remove.")
            return 0
        for row in rows:
            print(f"  removing id={row[0]} symbol={row[1]} ts={row[2]}")
        conn.execute(
            r"DELETE FROM decision_points WHERE symbol LIKE 'TEST\_%' ESCAPE '\'"
        )
        # Also clean the tracking table, which has FK by decision_id.
        conn.execute(
            r"DELETE FROM decision_tracking WHERE decision_id NOT IN "
            r"(SELECT id FROM decision_points)"
        )
        conn.commit()
        print(f"Removed {len(rows)} TEST_ rows.")
    finally:
        conn.close()
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

```python
"""One-shot cleanup of orphan Pending rows in the production DB.

An orphan is a Pending row that has been superseded by a completed
(non-Pending) row for the same symbol on the same day. Safe to re-run."""

import os
import sqlite3
import sys

DB_PATH = os.environ.get("DB_PATH", "subscribers.db")


def main() -> int:
    if not os.path.exists(DB_PATH):
        print(f"DB not found at {DB_PATH}; nothing to clean.")
        return 0
    conn = sqlite3.connect(DB_PATH)
    try:
        # Orphan Pendings: same (symbol, date) has a non-Pending sibling.
        cur = conn.execute(
            """
            SELECT id, symbol, recommendation, status, timestamp FROM decision_points p1
            WHERE status = 'Pending'
              AND EXISTS (
                  SELECT 1 FROM decision_points p2
                  WHERE p2.symbol = p1.symbol
                    AND date(p2.timestamp) = date(p1.timestamp)
                    AND p2.status != 'Pending'
                    AND p2.id != p1.id
              )
            """
        )
        orphans = cur.fetchall()

        # Stuck Pendings: only one Pending row for a (symbol, date) AND date < today.
        # Treat as crash residue; mark status='Failed' so it's visible but doesn't
        # block re-analysis.
        cur = conn.execute(
            """
            SELECT id, symbol, timestamp FROM decision_points p1
            WHERE status = 'Pending'
              AND date(timestamp) < date('now')
              AND NOT EXISTS (
                  SELECT 1 FROM decision_points p2
                  WHERE p2.symbol = p1.symbol
                    AND date(p2.timestamp) = date(p1.timestamp)
                    AND p2.id != p1.id
              )
            """
        )
        stuck = cur.fetchall()

        for row in orphans:
            print(f"  orphan: id={row[0]} {row[1]} {row[4]}")
        for row in stuck:
            print(f"  stuck:  id={row[0]} {row[1]} {row[2]}")

        if orphans:
            ids = [r[0] for r in orphans]
            conn.execute(
                f"DELETE FROM decision_points WHERE id IN ({','.join('?' * len(ids))})",
                ids,
            )
        if stuck:
            ids = [r[0] for r in stuck]
            conn.execute(
                f"UPDATE decision_points SET status = 'Failed', recommendation = 'PASS_CRASH' "
                f"WHERE id IN ({','.join('?' * len(ids))})",
                ids,
            )
        conn.commit()
        print(f"Removed {len(orphans)} orphan Pendings; marked {len(stuck)} stuck Pendings as Failed.")
    finally:
        conn.close()
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
